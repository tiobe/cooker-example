public class CookerPlate {
    private String TemperaturePrivate = "cold";

    public String Temperature() {
        return TemperaturePrivate;
    }

    public void heat() {

        if (TemperaturePrivate.equals("cold")) {
            TemperaturePrivate = "warm";
        } else {
            TemperaturePrivate = "hot";
        }
    }

    public void cool() {

        if (TemperaturePrivate.equals("hot")) {
           TemperaturePrivate = "Warm";
        } else{
            TemperaturePrivate = "cold";
        }
    }
}
